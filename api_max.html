Maximum

Description

Returns the maximum of the arguments. Defined over \(ℝ × ℝ\).

C base API (generic)

#define vmax(a0, a1, type)
#define vmax_e(a0, a1, type, simd_ext)

C++ base API (generic)

template <NSIMD_CONCEPT_VALUE_TYPE T> typename simd_traits<T, NSIMD_SIMD>::simd_vector NSIMD_VECTORCALL max(typename simd_traits<T, NSIMD_SIMD>::simd_vector a0, typename simd_traits<T, NSIMD_SIMD>::simd_vector a1, T);

C++ advanced API

template <NSIMD_CONCEPT_VALUE_TYPE T, NSIMD_CONCEPT_SIMD_EXT SimdExt> pack<T, 1, SimdExt> max(pack<T, 1, SimdExt> const& a0, pack<T, 1, SimdExt> const& a1);
template <NSIMD_CONCEPT_VALUE_TYPE T, int N, NSIMD_CONCEPT_SIMD_EXT SimdExt> pack<T, N, SimdExt> max(pack<T, N, SimdExt> const& a0, pack<T, N, SimdExt> const& a1);
