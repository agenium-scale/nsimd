Load array of structure

Description

Load array of structures of 3 members from unaligned memory.

C base API (generic)

#define vload3u(a0, type)
#define vload3u_e(a0, type, simd_ext)

C advanced API (generic, requires C11)

#define nsimd_load3u(a0)

C++ base API (generic)

template <NSIMD_CONCEPT_VALUE_TYPE T> typename simd_traits<T, NSIMD_SIMD>::simd_vectorx3 NSIMD_VECTORCALL load3u(T const* a0, T);

C++ advanced API

template <NSIMD_CONCEPT_VALUE_TYPE T, NSIMD_CONCEPT_SIMD_EXT SimdExt> packx3<T, 1, SimdExt> load3u(packx3<T, 1, SimdExt> const&, T const* a0);
template <NSIMD_CONCEPT_VALUE_TYPE T, int N, NSIMD_CONCEPT_SIMD_EXT SimdExt> packx3<T, N, SimdExt> load3u(packx3<T, N, SimdExt> const&, T const* a0);
template <NSIMD_CONCEPT_PACKX3 SimdVector, NSIMD_CONCEPT_VALUE_TYPE T>NSIMD_REQUIRES((std::is_same_v<typename SimdVector::value_type, T>)) SimdVector load3u(T const* a0);
