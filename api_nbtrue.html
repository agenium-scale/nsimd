Count true elements

Description

Return the number of true elements in the input.

C base API (generic)

#define vnbtrue(a0, type)
#define vnbtrue_e(a0, type, simd_ext)

C++ base API (generic)

template <NSIMD_CONCEPT_VALUE_TYPE T> int NSIMD_VECTORCALL nbtrue(typename simd_traits<T, NSIMD_SIMD>::simd_vectorl a0, T);

C++ advanced API

template <NSIMD_CONCEPT_VALUE_TYPE T, NSIMD_CONCEPT_SIMD_EXT SimdExt> int nbtrue(packl<T, 1, SimdExt> const& a0);
template <NSIMD_CONCEPT_VALUE_TYPE T, int N, NSIMD_CONCEPT_SIMD_EXT SimdExt> int nbtrue(packl<T, N, SimdExt> const& a0);
