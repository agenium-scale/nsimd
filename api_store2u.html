Store2u

Description

Store 2 SIMD vectors as array of structures of 2 members into unaligned memory.

C base API (generic)

#define vstore2u(a0, a1, a2, type)
#define vstore2u_e(a0, a1, a2, type, simd_ext)

C advanced API (generic, requires C11)

#define nsimd_store2u(a0, a1, a2)

C++ base API (generic)

template <NSIMD_CONCEPT_VALUE_TYPE T> void NSIMD_VECTORCALL store2u(T* a0, typename simd_traits<T, NSIMD_SIMD>::simd_vector a1, typename simd_traits<T, NSIMD_SIMD>::simd_vector a2, T);

C++ advanced API

template <NSIMD_CONCEPT_VALUE_TYPE T, NSIMD_CONCEPT_SIMD_EXT SimdExt> void store2u(T* a0, pack<T, 1, SimdExt> const& a1, pack<T, 1, SimdExt> const& a2);
template <NSIMD_CONCEPT_VALUE_TYPE T, int N, NSIMD_CONCEPT_SIMD_EXT SimdExt> void store2u(T* a0, pack<T, N, SimdExt> const& a1, pack<T, N, SimdExt> const& a2);
