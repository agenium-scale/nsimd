Reinterpret vector

Description

Reinterpret input vector into a different vector type preserving all bits.

C base API (generic)

#define vreinterpret(a0, from_type, to_type)
#define vreinterpret_e(a0, from_type, to_type, simd_ext)

C++ base API (generic)

template <NSIMD_CONCEPT_VALUE_TYPE F, NSIMD_CONCEPT_VALUE_TYPE T> NSIMD_REQUIRES(sizeof_v<F> == sizeof_v<T>) typename simd_traits<T, NSIMD_SIMD>::simd_vector NSIMD_VECTORCALL reinterpret(typename simd_traits<F, NSIMD_SIMD>::simd_vector a0, F, T);

C++ advanced API

template <NSIMD_CONCEPT_PACK ToPackType, NSIMD_CONCEPT_VALUE_TYPE T, NSIMD_CONCEPT_SIMD_EXT SimdExt> NSIMD_REQUIRES((sizeof_v<typename ToPackType::value_type> == sizeof_v<T> && ToPackType::unroll == 1 && std::is_same_v<typename ToPackType::simd_ext, SimdExt>))ToPackType reinterpret(ToPackType const&, pack<T, 1, SimdExt> const& a0);
template <NSIMD_CONCEPT_PACK ToPackType, NSIMD_CONCEPT_VALUE_TYPE T, int N, NSIMD_CONCEPT_SIMD_EXT SimdExt> NSIMD_REQUIRES((sizeof_v<typename ToPackType::value_type> == sizeof_v<T> && ToPackType::unroll == N && std::is_same_v<typename ToPackType::simd_ext, SimdExt>))ToPackType reinterpret(ToPackType const&, pack<T, N, SimdExt> const& a0);
